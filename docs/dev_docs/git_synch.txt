git remote add coasting_lmuff_alias https://github.com/coasting-nc/LMUFFB
# This command adds a new remote repository to your local Git configuration.

git fetch coasting_lmuff_alias
# This command downloads all branches, tags, and commits from the "coasting" remote.

git merge coasting_lmuff_alias/main --allow-unrelated-histories
# merges the main branch from the "coasting" remote into your current branch. --allow-unrelated-histories - this flag is necessary because your local repo and the remote repo don't share a common Git history (since you copied rather than forked)

# Alternative: Hard Reset (⚠️ Destructive)
# If you want to completely replace your current branch's content with the remote's content (discarding any local changes you've made):
git reset --hard coasting/main

# After syncing
# you can verify your branch is up to date with:
git log --oneline -10
